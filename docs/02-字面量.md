# 第二章：字面量

## 2.1 整数字面量

Jasper 支持多种进制整数，允许下划线 `_` 分隔提升可读性：

```jasper
function main(): int {
    var a: int = 42;            // 十进制
    var b: int = 0xFF;          // 十六进制 0x / 0X
    var c: int = 0o77;          // 八进制 0o / 0O
    var d: int = 0b1010;        // 二进制 0b / 0B

    var e: int = 1_000_000;     // 带下划线的十进制
    var f: int = 0xFF_FF;       // 十六进制也可用下划线

    return 0;
}
```

## 2.2 浮点数字面量

```jasper
function main(): int {
    var a: float = 3.14;        // 标准小数
    var b: float = .5;          // 省略整数部分
    var c: float = 1.0e10;      // 科学计数法
    var d: float = 1e-5;        // 负指数
    var e: float32 = 3.14f;     // 类型后缀 f / F
    var f: float64 = 3.14d;     // 类型后缀 d / D

    var g: float = 0x1.0p10;    // 十六进制浮点数 0x / 0X

    return 0;
}
```

## 2.3 字符串字面量

```jasper
function main(): int {
    // 普通字符串
    var s1: string = "Hello, Jasper!";

    // 转义字符
    var s2: string = "line1\nline2\tindented";

    // 原始字符串（反引号，内容保持原样）
    var s3: string = `C:\Users\name\file.txt`;

    // 正则字符串 r"..."
    var s4: regex = r"\d{3}-\d{4}";

    // 格式化字符串 f"..."（支持插值 {expr}）
    var name: string = "World";
    var s5: string = f"Hello, {name}!";

    // 字节字符串 b"..."
    var s6: bytes = b"\x00\x01\x02";

    // Unicode 字符串 u"..."
    var s7: string = u"\u4e2d\u6587";  // "中文"

    // 三引号字符串（支持多行）
    var s8: string = """
        这是多行字符串
        可以跨多行
        缩进和换行保留
        """;

    // 原始三引号字符串
    var s9: string = r"""
        原始多行字符串
        转义序列 \n 不会被处理
        """;

    // 格式化三引号字符串
    var s10: string = f"""
        多行插值：{name}
        计算：{1 + 2}
        """;

    return 0;
}
```

## 2.4 布尔与 null

```jasper
function main(): int {
    var t: bool = true;
    var f: bool = false;

    var n: int? = null;     // null 表示空值

    if (t) {
        return 1;
    }
    return 0;
}
```

## 2.5 dict 字面量

Jasper 使用 `dict{ ... }` 语法创建字典。key 可以是任意表达式：

```jasper
function main(): int {
    // 空字典
    var empty: dict<string, int> = dict{};

    // 基本键值对
    var scores: dict<string, int> = dict{
        "Alice": 95,
        "Bob": 87,
        "Charlie": 92
    };

    // key 支持任意表达式
    var prefix: string = "user_";
    var data: dict = dict{
        1: "one",
        prefix + "1": "first",
        true: "yes"
    };

    // 允许尾逗号
    var map: dict = dict{
        "a": 1,
        "b": 2,
    };

    return 0;
}
```

## 2.6 json 字面量

Jasper 使用 `json{ ... }` 创建 JSON 字面量。语法上 key/value 允许任意表达式（与 `dict` 一致），语义层保证结果为严格 JSON（key 运行时必须为 string，value 必须为 JSON 合法类型）。

```jasper
function main(): int {
    // 空 JSON
    var empty: json = json{};

    // 标准 JSON 对象
    var person: json = json{
        "name": "Alice",
        "age": 30,
        "active": true,
        "salary": null,
        "address": json{
            "city": "Beijing",
            "zip": "100000"
        },
        "tags": ["developer", "manager"]
    };

    // key/value 支持任意表达式（语义层确保结果符合 JSON）
    var key: string = "dynamic_key";
    var data: json = json{
        key: computeValue(),
        "static": 42,
    };

    // JSON 数组作为顶层值
    var list: json = json{
        "items": [1, 2, 3, 4, 5]
    };

    return 0;
}
```

## 2.7 数组初始化器

Jasper 使用 `[...]` 创建数组字面量，与数组类型 `int[]` 的语法呼应：

```jasper
function main(): int {
    // 数组字面量
    var arr: int[] = [1, 2, 3, 4, 5];

    // 带尾逗号
    var arr2: int[] = [
        1,
        2,
        3,
    ];

    return 0;
}
```

## 语法对照

| 特性 | Parser / Lexer 规则 | 对应代码 |
|------|---------------------|---------|
| 十进制整数 | `DEC_LITERAL` | `42`, `1_000_000` |
| 十六进制整数 | `HEX_LITERAL` | `0xFF`, `0xAB_CD` |
| 八进制整数 | `OCT_LITERAL` | `0o77`, `0o777` |
| 二进制整数 | `BIN_LITERAL` | `0b1010`, `0b11_00` |
| 浮点数 | `FLOAT_LITERAL` | `3.14`, `1e10`, `.5f` |
| 十六进制浮点数 | `HEX_FLOAT_LITERAL` | `0x1.0p10` |
| 普通字符串 | `STRING_LITERAL` | `"hello"` |
| 原始字符串 | `RAW_STRING_LITERAL` | `` `raw` `` |
| 正则字符串 | `REGEX_LITERAL` | `r"\d+"` |
| 格式化字符串 | `FORMAT_STRING_LITERAL` | `f"hello {name}"` |
| 字节字符串 | `BYTE_STRING_LITERAL` | `b"\x00"` |
| Unicode 字符串 | `UNICODE_STRING_LITERAL` | `u"\u4e2d"` |
| 三引号字符串 | `TRIPLE_STRING_LITERAL` | `"""..."""` |
| 原始三引号 | `RAW_TRIPLE_STRING_LITERAL` | `r"""..."""` |
| 格式化三引号 | `FORMAT_TRIPLE_STRING_LITERAL` | `f"""..."""` |
| 布尔 | `True` / `False` | `true`, `false` |
| null | `Null` | `null` |
| dict | `dictLiteral` | `dict{ "a": 1 }` |
| json | `jsonLiteral` | `json{ "a": 1 }` |
| 数组初始化 | `arrayInitializer` | `[1, 2, 3]` |

> 注：dict 的 key 是 `expression`（任意表达式），value 是 `expression`；
> json 语法上 key/value 也接受任意表达式，但语义层保证结果为严格 JSON（由 AST Checker 校验）。
